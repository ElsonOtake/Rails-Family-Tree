siblings
Leaf.find_by_sql("select name, gender from leafs join branches on branches.leaf_id = leafs.id join pairs on pairs.id = branches.pair_id where pairs.id = (select pair_id from branches where leaf_id = (select id from leafs where name = 'Elson Akio Otake')) and name <> 'Elson Akio Otake' order by birth")

parents
Leaf.find_by_sql("select name, gender from leafs where id in (select leafs.id from leafs join pairs on leafs.id = leaf1_id or leafs.id = leaf2_id join branches on branches.pair_id = pairs.id where branches.leaf_id = (select id from leafs where name = 'Elson Akio Otake'))")

partner
Leaf.find_by_sql("select name, gender from leafs where id in (select leaf1_id from pairs where leaf2_id = (select id from leafs where name = 'Elson Akio Otake') union select leaf2_id from pairs where leaf1_id = (select id from leafs where name = 'Elson Akio Otake'))")

children
Leaf.find_by_sql("select id, name, gender from leafs where id in (select leaf_id from branches join pairs on branches.pair_id = pairs.id where leaf1_id = (select id from leafs where name = 'Akira Otake') or leaf2_id = (select id from leafs where name = 'Akira Otake')) order by birth")
